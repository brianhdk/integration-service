Hi, and thanks for trying out the Swagger/NSwag extension.
To get started, modify your Program.cs as shown below to begin working with this.

Also make sure to visit the Project Home on https://github.com/vertica-as/integration-service for further documentation.

You can visit the NSwag project here: https://github.com/RSuter/NSwag/wiki

using Vertica.Integration.WebApi;
using Vertica.Integration.WebApi.NSwag;

namespace YourNameSpace
{
	class Program
	{
		static void Main(string[] args)
		{
			IntegrationStartup.Run(args, application => application
                .UseWebApi(webApi => webApi
                    .WithNSwag(nswag => nswag
                        .Title("Demo")
                        .Description("Show casing NSwag")
                        .DisableIf(condition => condition.IsProduction())
						.AddFromAssemblyOfThis<MyApiController>())));

		}
	}
}
